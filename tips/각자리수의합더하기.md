```
function solution(x) {
    //자릿수 다 더해서 
    let string = String(x)
    let sum = 0
    for(let i=0; i<string.length; i++){
        sum += parseInt(string[i])
    }
    //그 값으로 나누었을때 
    if(x%sum===0)return true
    else return false
}
```
