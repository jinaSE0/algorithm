```
function solution(s) {
    return s.split('').sort().reverse().join('')
}
//sort는 배열에 할 수 있는거 잊지말자구
//sort()를 할 경우, 자동으로 대문자가 먼저 A부터 Z로 정렬되고, 소문자가 a->z 로 정렬된다
```
